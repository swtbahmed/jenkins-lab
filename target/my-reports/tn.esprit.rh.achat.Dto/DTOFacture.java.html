package tn.esprit.rh.achat.Dto;

@Getter
@Setter

	private float montantRemise;
	private float montantFacture;
	private Date dateCreationFacture;
	private Date dateDerniereModificationFacture;

        this.detailsFacture = detailsFacture;

    }
}
